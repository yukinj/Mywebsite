class Solution:
    def rotate(self, matrix: List[List[int]]) -> None:
# time O(N^2) space O(1)
        n = len(matrix)
        # step1: transpose 
        for i in range(n):
            for j in range(i):
                matrix[i][j],matrix[j][i]=matrix[j][i],matrix[i][j]
        # step2: each row, reverse 
        for row in matrix:
            for j in range(n//2):
                row[j],row[~j] = row[~j],row[j]
