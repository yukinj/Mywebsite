```python
class Solution:
    def secondHighest(self, s: str) -> int:
        """tc O(N)  sc O(1)
        digits :  0 - 9 
        """
        fst, snd = -1, -1
        for c in s:
            
            if c.isdigit():
                x = int(c)
                if fst < x :
                    fst,snd = x, fst
                elif snd < x and x < fst: # prevent case x == fst
                    snd = x 
        return snd

```
