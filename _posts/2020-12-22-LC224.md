iterative  solution 
```python
class Solution:
    def calculate(self, s: str) -> int:
        res = 0
        num  = 0
        st = []
        sign = 1
        st.append(sign)
        for i in range(len(s)):
            if s[i].isdigit():
                num = num * 10 + int(s[i])
            if  s[i] in ['+','-'] or i == len(s)-1:
                res = res + sign * num
                sign = st[-1] * (1 if s[i]=='+' else -1)
                num = 0
            if s[i] == '(':
                st.append(sign)
            if s[i] == ')':
                st.pop()
        return res 
```
