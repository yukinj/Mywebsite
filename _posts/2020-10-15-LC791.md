class Solution:
    def customSortString(self, S: str, T: str) -> str:
        # time O(N^2) space O(N)
        cnt = collections.Counter(T)
        res = []
        for c in S:
            if c in cnt:
                res.append(c*cnt.pop(c))
        for c,v in cnt.items():
            res.append(c*v)
        return ''.join(res)
