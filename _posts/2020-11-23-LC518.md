class Solution:
    def change(self, amount: int, coins: List[int]) -> int:
        n = len(coins)
        # dp[i][j] previous coins there there are dp[i][j] ways to get j amount
        dp = [[0 for x in range(amount+1)] for _ in range(n+1)]
        # initialize: for target amount = 1, there is only 1 way to make up  
        for i in range(len(coins)+1):
            dp[i][0] = 1 
        for i in range(1,n+1):
            for j  in range(1,amount+1):
                if j - coins[i-1] >=0 :
                    dp[i][j] = dp[i-1][j] + dp[i][j-coins[i-1]]
                else:
                    dp[i][j] = dp[i-1][j]
        return dp[n][amount]
