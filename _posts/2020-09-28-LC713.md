class Solution:
    def numSubarrayProductLessThanK(self, nums: List[int], k: int) -> int:
        # time O(N) space O(1)
        if not nums or k <= 1 :return 0
        l  = cnt = 0
        mul = 1
        for r, val in enumerate(nums):
            mul *= val 
            while mul >= k and l <= r:
                mul //= nums[l]
                l += 1
            cnt += r - l + 1
        return cnt
