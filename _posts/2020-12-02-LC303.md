```python
class NumArray:

    def __init__(self, nums: List[int]):
        n = len(nums)
        self.psum = [0] * (n+1) 
        for i in range(n):
            self.psum[i+1] = self.psum[i] + nums[i]# psun[i]: sum of num[0]++num[i]

    def sumRange(self, i: int, j: int) -> int:
        return self.psum[j+1] - self.psum[i]
```
