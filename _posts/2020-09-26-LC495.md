class Solution:
    def findPoisonedDuration(self, timeSeries: List[int], duration: int) -> int:
        size = len(timeSeries)
        if size == 0:
            return 0
        res = 0
        for i in range(size-1):
            res += min(timeSeries[i+1] - timeSeries[i],duration)
        
        return res + duration
