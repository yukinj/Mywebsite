solution 1:

```python
class Solution:
    def rotate(self, nums: List[int], k: int) -> None:
# solution 1: create extra space and copy
# time O(N) space O(N)
        if not nums or k<=0:
            return 
        copy = []
        size = len(nums)
        for i in range(size):
            copy.append(nums[i])
        for j in range(size):
            nums[(j+k)%size] = copy[j]
```
