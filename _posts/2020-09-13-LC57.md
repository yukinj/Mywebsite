```python
class Solution:
    def insert(self, intervals: List[List[int]], newInterval: List[int]) -> List[List[int]]:
        """ time O(N) space: O(1)  
        main idea: use 3 loops to seperate different non-overlap inteverse
        edge case:  intvs is empty 
        case 1: intv end < new_start 
        case_2: intv_start <= new_end, note intv_start == new_end will meet overlap condition, don missout 
        case_3: intv_start > new_start 
    
        """
        if not intervals:
            return [newInterval]
        p = 0
        res = []
        n = len(intervals)
        case 1: intv end < new_start
        while p < n and intervals[p][1] < newInterval[0]:
            res.append([intervals[p][0],intervals[p][1]])
            p += 1 
        #case_2: intv_start <= new_end
        while p < n and intervals[p][0] <= newInterval[1]:  # overlap so don forget = 
            newInterval[0] = min(intervals[p][0],newInterval[0])
            newInterval[1] = max(newInterval[1],intervals[p][1])
            p += 1
        res.append([newInterval[0],newInterval[1]])
        
        #case_3: intv_start > new_start 
        while p < n and intervals[p][0] > newInterval[1]:
            res.append([intervals[p][0],intervals[p][1]])
            p += 1 
        return res 
```
