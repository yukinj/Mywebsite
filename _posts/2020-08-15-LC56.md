```python
class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
        """ tc O(NlgN)  sc O(1)
        1.sort by start 
        2.compare current with stack top, if overlap, combine ; otherwise, push into stack 
        3. return stack 
        """
        intervals.sort(key = lambda x: x[0])
        st = []
        for s,e in intervals:
            if st and st[-1][1] >= s:
                st[-1] = [st[-1][0],max(st[-1][1],e)] #note,need to update the biggest end time 
            else:
                st.append([s,e])
        return st 
```
