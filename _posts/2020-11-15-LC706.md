class ListNode:
    def __init__(self,key,val):
        self.pair =(key,val)
        self.next = None
        
class MyHashMap:

    def __init__(self):
        """
        Initialize your data structure here.
        """
        # mode for hash, table  
        self.m = 1000
        self.table = [None] * self.m

    def put(self, key: int, value: int) -> None:
        """
        value will always be non-negative.
        """
        # step1, get hcode 
        hc = key%self.m
        prev = None 
        cur = self.table[hc]
        # case 1: map[idx] == None;  case2: map[idx] != None
        if cur == None:
            self.table[hc] = ListNode(key,value)
        else:
            while cur:
                # find same key, update 
                if cur.pair[0] == key:
                    cur.pair = (key,value)
                    return
                prev = cur 
                cur = cur.next
            # same key not found, add new k,v with cached prev reference 
            prev.next = ListNode(key,value)
            
            

    def get(self, key: int) -> int:
        """
        Returns the value to which the specified key is mapped, or -1 if this map contains no mapping for the key
        """
        hc = key%self.m
        cur = self.table[hc]
        while cur:
            if cur.pair[0] == key:
                return cur.pair[1]
            cur = cur.next
        return -1 
        

    def remove(self, key: int) -> None:
        """
        Removes the mapping of the specified value key if this map contains a mapping for the key
        """
        hc = key%self.m
        cur = self.table[hc]
        # prev is to connect prev node with next node of cur 
        prev = None
        
        # big picture: keep look for  pair[0] == key  by cur = cur.next and cache prev 
        while cur:
            if cur.pair[0] == key: 
                # check if first node need to be removed, if yes ,assign next to table[hc] directly
                if prev == None:
                    self.table[hc] = cur.next # can be None if cur has no next node 
                    return 
                else:
                    # if middle or tail node need to be removed, if yes, connect prev with next 
                    prev.next = cur.next
                    return 
            prev = cur 
            cur = cur.next
         


# Your MyHashMap object will be instantiated and called as such:
