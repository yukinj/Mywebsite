```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def maxPathSum(self, root: TreeNode) -> int:
        maxSum = [root.val]
        
        def maxSubPath(node,maxSum):
            lm = rm = 0
            if node.left:
                lm = max(lm, maxSubPath(node.left,maxSum))
            
            if node.right:
                rm = max(rm, maxSubPath(node.right,maxSum))
                
            maxSum[0] = max(maxSum[0], lm + rm + node.val)
            return max(lm,rm) + node.val  # max subPath  
        maxSubPath(root,maxSum)
        return maxSum[0]

```
