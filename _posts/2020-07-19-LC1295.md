class Solution:
    def findNumbers(self, nums: List[int]) -> int:
        res = 0
        for num in nums:
            if  num > 9 and num < 100:
                res += 1
            if num > 999 and num < 9999:
                res += 1
            else:
                if num == 100000:
                    res += 1 
        return res
