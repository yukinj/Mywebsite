class Solution:
    def minIncrementForUnique(self, A: List[int]) -> int:
        """ time O(NlgN) space O(1)
        """
        low = res  =  0
        for a in sorted(A):
            if low > a :
                res += low - a  
            low = max(low + 1, a + 1)
        return res

class Solution:
